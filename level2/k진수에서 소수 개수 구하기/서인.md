const jinsu =(n,k)=>{
    let changed_number = ''
    while (n){
        changed_number = n % k + changed_number
        n = parseInt(n / k) 
    }
    return changed_number
}

const MAX_NUM = 1000001
// 에라토스테네스의 체로 1,000,000 이하의 수가 소수인지 아닌지 1번 구한다. 
// 메모리 많이 필요. 시간 복잡도는  O(Nlog(log N))
const isSosuArray= Array.from({length:MAX_NUM},()=>true)
for (let i = 2; i < Math.floor(Math.sqrt(MAX_NUM)+1); i++){ // 제곱근까지만 (1X8, 2X4, 4X2, 8X1 ..)
    if (isSosuArray[i]){
        for (let j = 2; i*j <= MAX_NUM; j++){
            isSosuArray[i*j] = false
        }
    }
}

//1은 소수아님
isSosuArray[1]=false


function solution(n, k) {
    const candidates = jinsu(n,k).split(0).map(num=>parseInt(num))
    let answer = 0;
    for (candidate of candidates){
        if (isSosuArray[candidate]) answer +=1
    }
    return answer;
}
